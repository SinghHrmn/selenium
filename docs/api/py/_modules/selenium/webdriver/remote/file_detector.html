[docs]    def is_local_file(self, *keys: AnyKey) -> Optional[str]:
        file_path = ''.join(keys_to_typing(keys))

        if not file_path:
            return None

        try:
            if os.path.isfile(file_path):
                return file_path
        except Exception:
            pass
        return None
